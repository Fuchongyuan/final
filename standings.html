// 🚀 最新的戰績表動態渲染引擎 (適用於新的 data.json 結構) function renderStandings() { const container = document.getElementById('standings-container'); container.innerHTML = ''; const standings = globalData.standings || {}; // 迴圈讀取 "美聯東區 (AL East)"、"國聯西區 (NL West)" 等分區 Object.keys(standings).forEach(divName => { const divCard = document.createElement('div'); divCard.className = 'division-card'; let rowsHTML = ''; standings[divName].forEach((team, index) => { rowsHTML += ` ${index + 1} ${team.code} (${team.name}) ${team.w} ${team.l} ${team.gb} ${team.l10} ${team.home} ${team.away} `; }); // 建立該分區的表格 HTML divCard.innerHTML = `
${divName}
${rowsHTML}
# 球隊 勝 敗 勝差 近10場 主場 客場
`; container.appendChild(divCard); }); }
